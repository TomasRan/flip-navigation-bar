## Install

```
# use npm
$ npm install flip-navigation-bar

# use bower
$ bower install flip-navigation-bar
```

## Usage

```
	var FlipNavBar = require('flip-navigation-bar');

	var flipNavBar = new FilpNavBar({
		'container': '.main',
		'navBarWidth': '',
		'navBtnWidth': '',
		'navItemWidth': '',
		'defaultSelected': '3',
		'navData': [],
		'statusClassMap': {},
		'jumpToSelected': false,
		'calllback': function() {}
		'preBtn': {},
		'nextBtn': {}
	});
```
